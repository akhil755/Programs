import java.util.Scanner;
import java.util.Map;
import java.util.TreeMap;
import java.util.stream.Collectors;
import java.util.Arrays;
import java.util.List;
import java.util.function.Function;

public class FindDuplicateWordsInASentenceUsingStreams{
    public static void main(String[] args){
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter a sentence");
        String sentence = scanner.nextLine();
        
        List<String> words = Arrays.stream(sentence.split("\\W+"))
            .map(String::toLowerCase)
            .collect(Collectors.toList());
            
            Map<String, Long> duplicates = words.stream()
            .collect(Collectors.groupingBy(Function.identity(), TreeMap::new, Collectors.counting()))
            .entrySet()
            .stream()
            .filter(entry-> entry.getValue()>1)
            .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));
            
            if(duplicates.isEmpty()){
                System.out.println("no duplicate words present in "+sentence);
            }else{
                duplicates.forEach((word,count)->System.out.println(word+"->"+count+" times "));
            }
    }
}
